Lütfen seçtiğiniz algoritmaya ait çözümü ve diğer isterleri aşağıya ekleyiniz:

ALGORITMA MukemmelOmletYap

BAŞLA
    // 1. Gerekli malzemeleri tanımla
    yumurta_sayisi ← KULLANICIDAN_AL("Kaç yumurta kullanacaksınız?")
    süt_miktari ← 30 ml        // opsiyonel
    tuz_miktari ← 1 çay kaşığı
    karabiber_miktari ← 1 çay kaşığı
    tereyağı_miktari ← 1 yemek kaşığı
    tava_sıcaklığı ← 0
    pişirme_süresi ← 3 dakika

    // 2. Yumurta sayısı kontrolü
    EĞER yumurta_sayisi < 2 İSE
        YAZDIR("Omlet için en az 2 yumurta gerekir.")
        yumurta_sayisi ← 2
    SON

    // 3. Yumurtaları kır ve kaseye ekle
    KARISIM ← BOŞ_KASE_OLUŞTUR()
    DÖNGÜ i ← 1'DEN yumurta_sayisi'NE KADAR
        YUMURTA_KIR()
        KARISIMA_EKLE("yumurta")
    SON_DÖNGÜ

    // 4. Opsiyonel süt ekleme
    EĞER süt_miktari > 0 İSE
        KARISIMA_EKLE(süt_miktari)
    SON

    // 5. Baharat ekleme
    KARISIMA_EKLE(tuz_miktari)
    KARISIMA_EKLE(karabiber_miktari)

    // 6. Karışımı homojen hale gelene kadar çırp
    karışım_homojen ← FALSE
    WHILE karışım_homojen = FALSE
        ÇIRP(KARISIM)
        EĞER HOMOJEN_KONTROL(KARISIM) = TRUE İSE
            karışım_homojen ← TRUE
        SON
    SON_WHILE

    // 7. Tavayı ısıt
    TAVAYI_OCAĞA_KOY()
    ATEŞI_AYARLA("orta")
    TAVAYA_EKLE(tereyağı_miktari)

    WHILE tava_sıcaklığı < 160°C
        tava_sıcaklığı ← SICAKLIK_ÖLÇ()
    SON_WHILE
    YAZDIR("Tava hazır, karışımı dökebilirsiniz.")

    // 8. Karışımı tavaya dök
    TAVAYA_DÖK(KARISIM)

    // 9. Pişirme süreci
    pişirme_başlangıç ← ZAMAN_ŞİMDİ()
    WHILE GECEN_SÜRE(pişirme_başlangıç) < pişirme_süresi
        EĞER KENARLAR_PIŞTİ_MI() = TRUE VE ORTA_KISIM_SIVI() = TRUE İSE
            SPATULA_İLE_KENARLARI_KALDIR()
        SON
    SON_WHILE

    // 10. Katlama ve son kontrol
    EĞER OMLET_HAZIR_MI() = TRUE İSE
        OMLETİ_KATLA()
        YAZDIR("Omletiniz hazır! Afiyet olsun.")
    DEĞİLSE
        YAZDIR("Omlet tam pişmedi, biraz daha bekletin.")
    SON

BİTİR
